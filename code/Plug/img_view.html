    //滚轮放大图片

    function bigimg(i) {

    var zoom = parseInt(i.style.zoom, 10) || 100;

    zoom += event.wheelDelta / 12;

    if (zoom > 0)

    i.style.zoom = zoom + '%';

    return false;

    }

    </script>
