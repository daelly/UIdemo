function hideload() {

1、需要确认的提示：rsMessageBox.confirm(“标题”,”提示内容”,function(){});
rsMessageBox.confirm("提示","确定要删除当前数据吗？" ,function(){
	jQuery.ajax({
		url : "${contextPath}/FiStaffDetail.mc?method=deleteFiStaffDetail",
		type : "post",
		dataType:"json",
		data : {PK:PK,staffId:staffId},
		success : function(result) {
				top.rsMessageBox.ok("删除成功！");
			} else {
				top.rsMessageBox.info(result.meg);
			}
		},
		error : function(result) {
			top.rsMessageBox.alert("删除失败");
		}
	});
});
2、信息性消息rsMessageBox.info(“标题”,”提示内容”);
3、成功消息rsMessageBox.ok(“标题”,”提示内容”);
4、警告消息rsMessageBox.alert(“标题”,”提示内容”);
5、错误、失败消息rsMessageBox.error(“标题”,”提示内容”);
6、提示消息rsMessageBox.tips(“标题”,”提示内容”);
7、弹窗提示rsMessageBox.alertmodal“标题”,”提示内容”,function(){});
8、自定义提示消失时间：（以上除了确认和弹窗，其余的消息默认2500毫秒后消失）
rsMessageBox.show('<h4>'+ title +'</h4>'+ message +'！',type, time);
Type可选值：info，success，warning，error
Time：3000
实例：rsMessageBox.show('<h4>'+ title +'</h4>'+ message +'！',’info’, 4000);
