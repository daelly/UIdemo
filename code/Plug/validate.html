表单验证所需引入js(前提是已经引入boostrap) 按需加载所需的验证js：

<script>

if (typeof (requreLib) != "undefined") {

requreLib.setplugs("jValidate");

}

</script>


使用方式：在input、textarea等输入框元素中添加属性validate, 属性值按需填写，共有如下几种(多种按,分隔)。 然后在js方法中（如下所示的checkForm()中进行验证）。

<script type="text/javascript">

function checkForm() {

//验证

var isValidation = $("#testForm").validate();

//如果isValidation为true,则表示验证通过，否则验证不通过

if (isValidation) {

}

}

</script>


所有可验证项： 
required: "必输项",
email: "请输入有效的电子邮件地址，格式如：user@domain.com",
url: "请输入有效的URL地址，格式如： http://www.domain.com",
date: "无效的日期，格式必需为 YYYY-MM-DD",
number: "请输入有效数值",
plusnumber: "请输入正整数数值",
plusnumber1: "请输入正数",
creditcard: "请输入有效的信用卡号",
isCardNo: "身份证号输入错误: 长度应为15位或18位!",
phone:"请输入有效的电话号码",
english:"只能输入英文",
noChina:"只能输入大小写字母，数字，-，_",
special:"不接受特殊字符",
maxLength: "该输入项的最大长度是{0}字符",(例如果最大长度为20，则validate的属性值为maxLength[20],即validate="maxLength[20]")
minLength: "该输入项的最小长度是{0}字符",(例如果最小长度为20，则validate的属性值为minLength[20],即validate="minLength[20]")
rangeLength: "该输入项的长度范围应该在{0}字符之间",(例如果长度范围为20，则validate的属性值为rangeLength[20],即validate="rangeLength[20]")
confirmPassWord:"两次输入的密码不一样",
maxValue: "该输入项的最大值是{0}",(例如果最大值为20，则validate的属性值为maxValue[20],即validate="maxValue[20]")
minValue: "该输入项的最小值是{0}"(例如果最小值为20，则validate的属性值为minValue[20],即validate="minValue[20]")
